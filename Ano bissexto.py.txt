from datetime import date

ano = int(input("Qual ano voce quer analisar?\033[1;36mColoque 0 para analisar o ano atual:\033[m"))


if ano == 0:
    ano = date.today().year




if ano % 4 == 0 and ano % 100 != 0 or ano % 400 == 0:
    print("O ano \033[1;31m{}\033[m é Bissexto".format(ano))
else:
    print("O ano \033[4;33m{}\033[m nao é Bissexto".format(ano))
